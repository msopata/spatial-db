-- 6)
	-- a)
		SELECT sum(ST_Length(drogi.geometria)) AS dlugosc_drog FROM drogi;
	-- b)
		SELECT ST_AsText(geometria), ST_AREA(geometria), ST_Perimeter(geometria) 
		FROM budynki WHERE nazwa LIKE 'BuildingA';
	-- c)
		SELECT nazwa, ST_AREA(geometria) FROM budynki 
		GROUP BY nazwa, geometria ORDER BY nazwa;
	-- d)
		SELECT nazwa, ST_Perimeter(geometria) FROM budynki 
		GROUP BY nazwa, ST_Perimeter(geometria),  ST_AREA(geometria) 
		ORDER BY ST_AREA(geometria) DESC LIMIT 2;
	-- e)
		SELECT ST_Distance(a.geometria, b.geometria) 
		FROM budynki a, punkty_informacyjne b
		WHERE a.nazwa LIKE 'BuildingC' AND b.nazwa LIKE 'G';
	-- f)
		SELECT ST_AREA( ST_Difference(a.geometria, ST_Buffer(b.geometria, 0.5, 'join=mitre side=left'))) 
		FROM budynki a, budynki b WHERE a.nazwa='BuildingC' AND b.nazwa='BuildingB';
	-- g)
		SELECT budynki.nazwa FROM budynki, drogi 
		WHERE ST_Y(ST_Centroid(budynki.geometria)) > ST_Y(ST_PointN(drogi.geometria,1)) 
		AND drogi.nazwa = 'RoadX';
